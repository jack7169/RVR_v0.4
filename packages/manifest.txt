#   libopenssl3_*.ipk
#   liblzo2_*.ipk
#   zlib_*.ipk
#   git_*.ipk
#   git-http_*.ipk
#   libcurl4_*.ipk
#   ca-bundle_*.ipk
#
# gcs/ - GCS-only packages
#   kcptun-server_*.ipk
#   sshpass_*.ipk
#
# aircraft/ - Aircraft-only packages
#   kcptun-client_*.ipk
#
# NOTE: tinc is NOT included — replaced by l2tap static binary
#
